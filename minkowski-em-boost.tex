    % light cone
    \draw[light cone] (-\xrange,\xrange) -- (O);

    % triangles
    \draw[anglemarker,red,samples=\Nsamples,smooth,variable=\x,domain=0:0.4] plot({0.5 * sinh(\x)},{0.5 * cosh(\x)}) node[text=black] at (0.07,0.4) {$\zeta$};

    \draw[anglemarker,red,samples=\Nsamples,smooth,variable=\x,domain=0.4:0.8] plot({0.5 * sinh(\x)},{0.5 * cosh(\x)}) node[text=black] at (0.22,0.45) {$\zeta$};
    \fill[red,opacity=0.05,thick,samples=\Nsamples,smooth,variable=\x,domain=0.4:0.8] plot({0.5 * sinh(\x)},{0.5 * cosh(\x)}) -- (O);

    \draw plot [mark=*, mark size=0.1] coordinates{({0.95 * sinh(0.4) + cosh(0.4)*0.05}, {0.95 * cosh(0.4) + sinh(0.4)*0.05})};


    \draw[anglemarker,red,samples=\Nsamples,smooth,variable=\x,domain=0:0.4] plot({0.5 * cosh(\x)},{0.5 * sinh(\x)}) node[text=black] at (0.45,0.09) {$\zeta$};
    \fill[red,opacity=0.05,thick,samples=\Nsamples,smooth,variable=\x,domain=0:0.4] plot({0.5 * cosh(\x)},{0.5 * sinh(\x)}) -- (O);
